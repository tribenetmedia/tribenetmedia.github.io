<!DOCTYPE html>

<html>

<head>

    <title>Access Vercel Function</title>

</head>

<body>

    <div id="response-container">

        <!-- The response will be displayed here -->

    </div>


    <script>

        // Define the URL of your Vercel function

        const functionUrl = "https://your-project-name.vercel.app/api/yourFunctionName";


        // Function to make a GET request

        const fetchData = async () => {

            try {

                const response = await fetch(functionUrl);


                if (response.ok) {

                    const data = await response.json();

                    displayResponse(data);

                } else {

                    throw new Error(`HTTP error! Status: ${response.status}`);

                }

            } catch (error) {

                console.error("Fetch Error:", error);

                displayResponse({ error: "An error occurred while fetching data." });

            }

        };


        // Function to display the response

        const displayResponse = (data) => {

            const responseContainer = document.getElementById("response-container");


            if (data.error) {

                responseContainer.textContent = `Error: ${data.error}`;

            } else {

                responseContainer.textContent = JSON.stringify(data, null, 2);

            }

        };


        // Call the fetchData function when the page loads

        fetchData();

    </script>

</body>

</html>
